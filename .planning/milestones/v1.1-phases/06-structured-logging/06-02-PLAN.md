   c. After the existing option parsing block (after `retryDelayMs` is parsed, around line 190), add log level parsing:
      ```typescript
      const VALID_LOG_LEVELS = new Set(['debug', 'info', 'warn', 'error', 'fatal'] as const);
      const rawLogLevel = readOptionValue(args, '--log-level') ?? 'info';
      if (!VALID_LOG_LEVELS.has(rawLogLevel as DaemonLogLevel)) {
        writeError(io.stderr, `Invalid --log-level "${rawLogLevel}". Valid: debug, info, warn, error, fatal`);
        return 1;
      }
      const logLevel = rawLogLevel as DaemonLogLevel;
      ```

   d. Import `createDaemonLogger` and `DaemonLogLevel` from `../daemon/logger.js`.

   e. Create the logger: `const logger = createDaemonLogger(logLevel);`

      ```typescript
      const daemon = await startDaemon({
        configPath: path.resolve(io.cwd, configPath),
        repositoryRoot: io.cwd,
        registerSignalHandlers: false,
        ...(monitor ? { monitor } : {}),
        logger
      });
      ```
      Note: logger is ALWAYS passed (not conditional) since `createDaemonLogger` defaults to 'info'. This ensures structured logging is always active when using the worker command.
